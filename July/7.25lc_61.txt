/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* deleteDuplicates(ListNode* head) {
        if(!head||!head->next)
            return head;
        ListNode* fakeHead=new ListNode(-1);
        fakeHead->next=head;
        ListNode* p=fakeHead;
        while(p&&p->next){
            ListNode* sameP=p;
            int count=0;
            while(sameP->next&&sameP->next->val==p->next->val){
                sameP=sameP->next;
                count++;
            }
            if(count>1)
                p->next=sameP->next;
            else
                p=sameP;
        }
        return fakeHead->next;
    }
};
