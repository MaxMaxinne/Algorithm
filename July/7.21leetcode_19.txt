/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeNthFromEnd(ListNode* head, int n) {
        ListNode* pFast=head;
        ListNode* pSlow=head;
        for(int i=0;i<n-1;i++)
            pFast=pFast->next;
        if(pFast->next==nullptr)
            return head->next;
        else
            pFast=pFast->next;
        while(pFast->next){
            pFast=pFast->next;
            pSlow=pSlow->next;
        }
        pSlow->next=pSlow->next->next;
        return head;
    }
};
