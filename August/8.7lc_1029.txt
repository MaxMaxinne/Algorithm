class Solution {
public:
    int twoCitySchedCost(vector<vector<int>>& costs) {
        vector<int> price;
        int cost=0;
        for(int i=0;i<costs.size();i++){
            cost+=costs[i][0];
            price.push_back(costs[i][1]-costs[i][0]);
        }
        sort(price.begin(),price.end());
        for(int i=0;i<costs.size()/2;i++)
            cost+=price[i];
        return cost;
    }
};
