class Solution {
public:
    vector<int> plusOne(vector<int>& digits) {
        int last=digits.size()-1;
        digits[last]++;
        for(int i=last;i>=0;i--){
            if(digits[i]>=10){
                digits[i]-=10;
                if(i-1>=0)
                    digits[i-1]++;
                else
                    digits.insert(digits.begin(),1);
            }
                else
                    break;
            }
        return digits;
        }
};
